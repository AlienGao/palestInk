# 给定一个非负整数 numRows，生成杨辉三角的前 numRows 行。

# 在杨辉三角中，每个数是它左上方和右上方的数的和。

示例:

输入: 5
输出:
[
     [1],
    [1,1],
   [1,2,1],
  [1,3,3,1],
 [1,4,6,4,1]
]

解答:

/**
 * @param {number} numRows
 * @return {number[][]}
 */
var generate = function(numRows) {
    var result = [[1]]
    if (numRows === 0) return []
    if (numRows === 1) return result
    if (numRows === 2) {
        result = [[1],[1,1]]
        return  result
    } else {
        result = [[1],[1,1]]
        while(result.length < numRows) {
            result.push([])
            result[result.length-1][0] = 1
            result[result.length-1][result.length-1] = 1
            for (let i = 1; i < result.length - 1; i++) {
                result[result.length-1][i] = result[result.length-2][i - 1] + result[result.length-2][i]
            }
        }        
    }
    return result
};
