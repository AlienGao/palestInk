# 给定一个非负整数数组 A，返回一个由 A 的所有偶数元素组成的数组，后面跟 A 的所有奇数元素。

# 你可以返回满足此条件的任何数组作为答案。

示例:

输入：[3,1,2,4]
输出：[2,4,3,1]
输出 [4,2,3,1]，[2,4,1,3] 和 [4,2,1,3] 也会被接受。

示例:

  1. 1 <= A.length <= 5000
  2. 0 <= A[i] <= 5000

解答:

/**
 * @param {number[]} A
 * @return {number[]}
 */
var sortArrayByParity = function(A) {
    var binary = []
    var parity = []
    for (let i = 0; i < A.length; i++) {
        if (A[i] % 2 !== 0) {
            binary.push(A[i])
        } else {
            parity.push(A[i])
        }
    }
    return parity.concat(binary)
};
